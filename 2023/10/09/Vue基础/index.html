Vue 基础

插值 v-bind

v-bind所绑定的数据是不可以影响 data 内部的变量的，但data内部的变量变化时，可以影响v-bind所绑定的内容。

 <div id="app">
      <!-- 01-静态文本 -->
      <h1>你好！明昊</h1>
      <!-- 02-动态文本 -->
      <h1>你好！{{name}}</h1>   // 普通插值  结果为  你好！ 明昊
     
      <a v-bind:href="link">链接</a> //单项数据绑定
   
      <a :href="link">链接</a> //单项数据绑定 简写   
    
   </div>

const app = Vue.createApp({
  data() {
    return {
      name: "眀昊"
      link: "https://baidu.com",
    };
  },
});
app.mount("#app");

V-for

<div id="app">
    <ul>
        <li v-for="todo in todos">{{ todo }}</li>   //渲染 todos 数组元素  todo自己起的名字
        <li v-for="(todo, index) in todos">{{index + 1}}. {{ todo }}</li> //index 是索引 就是 123456 这些  可以给index+1
        <li v-for="todo in todos">
          <input type="checkbox" :checked="todo.complete" />{{todo.content}}   // 渲染对象类型的  
        </li>
      </ul>
</div>

const app = Vue.createApp({
  data() {
    return {
      // todos: ["把项目做完", "去超市购物", "看10分钟的书"],
      todos: [
        {
          content: "把项目做完",
          complete: true,
        },
        {
          content: "去超市购物",
          complete: false,
        },
        {
          content: "看10分钟的书",
          complete: false,
        },
      ],
    };
  },
});
app.mount("#app");

V-show 和 v- if

v-show 和v-if 都是条件渲染 真时显示元素 假时隐藏元素 用法差不多一样

区别就是 v-show 不显示 是让其display属性为none v-if 直接把代码pass了

v-if还有v-else if 等等等等能一直连用

当v-if后面表达式的值为true的时候显示v-if所在的元素
当v-if后面表达式的值为false的时候显示v-else所在的元素

 <div id="app">
      <div class="question">
        <p>问：Vue 是一个什么样的框架？</p>
        <p v-show="showAnswer" class="answer">  //
          答：Vue 是一套用于构建用户界面的渐进式框架。
        </p>
        <button @click="showAnswer = !showAnswer">
          {{ showAnswer ? "隐藏" : "显示" }}答案
        </button>
          
    <p v-if="score>=90">优秀</p>
    <p v-else-if="score>=60">及格</p>
    <p v-else>不及格</p>
      </div>
    </div>

// 示例：事件处理和响应性入门
const app = Vue.createApp({
  data() {
    return {
      showAnswer: false,
      score: 50,
    };
  },
});
app.mount("#app");

当需要在显示与隐藏之间切换很频繁时, 使用v-show
当只有一次切换时, 使用 v-if

Vue Watch 监听器

watch是vue内部提供的一个用于侦听功能的更通用的方法，其用来响应数据的变化，通过特定的数据变化驱动一些操作。
vue官方文档解释当需要在数据变化时执行异步或开销较大的操作时，推荐使用该方法。

我的理解 就跟方法一样 不过方法名字是要监听的属性 这属性一变化 就执行 这里头的方法

    <div id="app">
      <div class="question">
        <p>问：Vue 是一个什么样的框架？</p>
        <p v-show="showAnswer" class="answer">
          答：Vue 是一套用于构建用户界面的渐进式框架。
        </p>
        <button @click="toggleAnswer">{{ label }}</button>
      </div>
    </div>

// 示例：使用 watch 监听数据变化
const app = Vue.createApp({
  data() {
    return {
      showAnswer: false,
      countDown: 5,
      timer: null,
    };
  },
  computed: {
    label() {
      return this.showAnswer ? "隐藏答案 " + this.countDown : "显示答案";
    },
  },
  methods: {
    toggleAnswer() {
      this.showAnswer = !this.showAnswer;
    },
  },
  watch: {
    showAnswer(newVal, oldVal) {      // 监听  showAnswer 属性 
      if (newVal) {
        this.countDown = 5;
        if (this.timer) {
          clearInterval(this.timer);
          this.timer = null;
        }
        // 5秒后关闭答案
        this.timer = setInterval(() => {
          this.countDown -= 1;
          if (this.countDown === 0) {
            this.showAnswer = false;
            clearInterval(this.timer);
            this.timer = null;
          }
        }, 1000);
      }
    },
  },
});
app.mount("#app");

v-model

v-model 双向数据绑定 是可以影响 data 内部的变量的，data内部的变量变化时，可以影响v-model所绑定的内容。 

 <div id="app">
      <div>
        <div class="input-field">
          <label for="username">用户名：</label>
          <input
            id="username"
            type="text"
            @input="handleInput"
            :value="username"
          />
          <button @click="resetUsername">重置</button>
        </div>
        <div class="result">
          <p>你输入的用户名是：{{ username }}</p>
        </div>
      </div>
    </div>

const app = Vue.createApp({
  data() {
    return {
      username: "",
    };
  },
  methods: {
    handleInput(e) {
      this.username = e.target.value;
    },
    resetUsername() {
      this.username = "";
    },
  },
});
app.mount("#app");

在 inline-style 内联样式中绑定 data 数据

 <div id="app">
      <div>
        <p :style="pStyle">这是一个段落</p>
        <button class="btn" @click="fontSize += 1">A</button>

const app = Vue.createApp({
  data() {
    return {
      fontSize: 16,
    };
  },
  computed: {
    pStyle() {
      return {
        color: "white",
        fontSize: this.fontSize + "px",
      };
    },
  },
});
app.mount("#app");
